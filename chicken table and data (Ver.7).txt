CREATE USER CHICKEN IDENTIFIED BY 123 DEFAULT TABLESPACE USERS;
GRANT CONNECT, RESOURCE, DBA TO CHICKEN;

DROP TABLE review PURGE;
DROP TABLE orderInfo PURGE;
DROP TABLE qna PURGE;
DROP TABLE userInfo PURGE;
DROP TABLE product PURGE;
DROP TABLE cart PURGE;

create table userInfo
(id VARCHAR2(10) CONSTRAINT USERINFO_ID_PK PRIMARY KEY,
pwd VARCHAR2(10) CONSTRAINT USERINFO_PWD_NN NOT NULL,
name VARCHAR2(10) CONSTRAINT USERINFO_NAME_NN NOT NULL,
email VARCHAR2(30) CONSTRAINT USERINFO_EMAIL_NN NOT NULL,
gender VARCHAR2(5) CONSTRAINT USERINFO_GENDER_CK CHECK(GENDER IN ('M','W','m','w','남자','여자','남','여') AND GENDER IS NOT NULL),
tel VARCHAR2(15) CONSTRAINT USERINFO_TEL_NN NOT NULL,
birth VARCHAR2(12) CONSTRAINT USERINFO_BIRTH_NN NOT NULL,
addr VARCHAR2(100) CONSTRAINT USERINFO_ADDR_NN NOT NULL,
created DATE CONSTRAINT USERINFO_CREATED_NN NOT NULL);

create table cart(
num number CONSTRAINT CART_NUM_NN NOT NULL,
id VARCHAR2(20) CONSTRAINT CART_ID_FK REFERENCES userInfo(id),
productName VARCHAR2(100) CONSTRAINT CART_PRODUCTNAME_NN NOT NULL,
qty NUMBER CONSTRAINT CART_QTY_NN NOT NULL,
price NUMBER CONSTRAINT CART_PRICE_NN NOT NULL,
pic1 VARCHAR2(100)CONSTRAINT CART_PIC1_NN NOT NULL);

create table product
(productNum VARCHAR2(5) CONSTRAINT PRODUCT_PRODUCTNUM_PK PRIMARY KEY,
category1 VARCHAR2(50) CONSTRAINT PRODUCT_CATEGORY1_NN NOT NULL,
category2 VARCHAR2(50) CONSTRAINT PRODUCT_CATEGORY2_NN NOT NULL,
category3 VARCHAR2(50) CONSTRAINT PRODUCT_CATEGORY3_NN NOT NULL,
productName VARCHAR2(100) CONSTRAINT PRODUCT_PRODUCTNAME_NN NOT NULL,
price NUMBER CONSTRAINT PRODUCT_PRICE_NN NOT NULL,
amount NUMBER CONSTRAINT PRODUCT_AMOUNT_NN NOT NULL,
qty NUMBER CONSTRAINT PRODUCT_QTY_NN NOT NULL,
pic1 VARCHAR2(200),
pic2 VARCHAR2(200),
pic3 VARCHAR2(200),
pic4 VARCHAR2(200),
CONSTRAINT PRODUCT_PRODUCTNAME_UK UNIQUE(productName));

create table orderInfo
(orderNum NUMBER CONSTRAINT ORDERINFO_ORDERNUM_PK PRIMARY KEY,
id VARCHAR2(10) CONSTRAINT ORDERINFO_ID_FK REFERENCES userInfo(id),
productNum VARCHAR2(5) CONSTRAINT ORDERINFO_PRODUCTNUM_FK REFERENCES product(productNum),
qty NUMBER CONSTRAINT ORDERINFO_QTY_NN NOT NULL,
buyerName VARCHAR2(10) CONSTRAINT ORDERINFO_BUYERNAME_NN NOT NULL,
buyerTel VARCHAR2(15) CONSTRAINT ORDERINFO_BUYERTEL_NN NOT NULL,
buyerEmail VARCHAR2(30) CONSTRAINT ORDERINFO_BUYEREMAIL_NN NOT NULL,
getterName VARCHAR2(10) CONSTRAINT ORDERINFO_GETTERNAME_NN NOT NULL,
getterTel1 VARCHAR2(15) CONSTRAINT ORDERINFO_GETTERTEL_NN NOT NULL,
getterTel2 VARCHAR2(15),
getterZip VARCHAR2(10) CONSTRAINT ORDERINFO_GETTERZIP_NN NOT NULL,
getterAddr VARCHAR2(100) CONSTRAINT ORDERINFO_GETTERADDR_NN NOT NULL,
getterMsg VARCHAR2(1000),
payWay VARCHAR(20) CONSTRAINT ORDERINFO_PAYWAY_NN NOT NULL,
payerName VARCHAR(10) CONSTRAINT ORDERINFO_PAYERNAME_NN NOT NULL,
bankName VARCHAR(100) CONSTRAINT ORDERINFO_BANKNAME_NN NOT NULL,
totalCost NUMBER CONSTRAINT ORDERINFO_TOTALCOST_NN NOT NULL,
orderDate DATE CONSTRAINT ORDERINFO_ORDERDATE_NN NOT NULL,
orderState VARCHAR2(10) CONSTRAINT ORDERINFO_ORDERSTATE_CK CHECK (orderState IN ('배송전','배송중','배송완료') AND orderState IS NOT NULL));

create table review(
reviewNum NUMBER CONSTRAINT REVIEW_REVIEWNUM_PK PRIMARY KEY,
id VARCHAR2(10) CONSTRAINT REVIEW_ID_FK REFERENCES userInfo(id),
orderNum NUMBER CONSTRAINT REVIEW_ORDERNUM_FK REFERENCES orderInfo(orderNum),
productName VARCHAR2(50) CONSTRAINT REVIEW_PRODUCTNAME_NN REFERENCES product(productName),
score NUMBER CONSTRAINT REVIEW_SCORE_CC CHECK (score BETWEEN 1 AND 5 AND score IS NOT NULL),
subject VARCHAR2(50) CONSTRAINTS REVIEW_SUBJECT_NN NOT NULL,
content VARCHAR2(1000) CONSTRAINTS REVIEW_CONTENT_NN NOT NULL,
created DATE CONSTRAINT REVIEW_CREATED_NOT NOT NULL);

create table qna(
qnaNum NUMBER CONSTRAINT QNA_QNANUM_PK PRIMARY KEY,
id VARCHAR2(50) CONSTRAINT QNA_ID_NN NOT NULL,
subject VARCHAR2(100) CONSTRAINTS QNA_SUBJECT_NN NOT NULL,
content VARCHAR2(1000) CONSTRAINTS QNA_CONTENT_NN NOT NULL,
created DATE CONSTRAINT QNA_CREATED_NOT NOT NULL);
